# 快速上手

注意

请确保你的 npm 版本或者 yarn 版本配置无误

# 使用 vue-cli@2.x

# 初始化项目
vue init acribus/acribus-webpack-boilerplate your-project-name

# done

如果不使用 acribus-cli 请继续阅读

# 直接使用 acribus

# 安装

yarn add acribus [--dev/-D] # 或者：npm install acribus -D

# 修改 webpack 配置

由于 acribus 发布的是源码，需要过 bable 编译

{
  test: /\.js$/,
  use: ['babel-loader'],
  exclude: /node_modules\/(?!(acribus|acribus-core)\/).*/
}

# 引入 acribus

在 main.js 中写入以下内容：

import Vue from 'vue'
import 'acribus-theme/dist/acribus/index.css'
import App from './App.vue'
import store from './store/index.js'
import router from './router/index.js'
import request from './api/request'

import acribus from 'acribus'
Vue.use(acribus, {
  request /*用于发起 Ajax 请求的对象*/
})

new Vue({
  el: '#app',
  router,
  store,
  render: h => h(App)
})

需要注意的是，样式文件需要单独引入。

# 声明页面所需要的资源

项目下建 resource 文件夹，用存放所有 resource 的描述对象。 比如页面包含 Person资源，新建 person.js：

import {schema} from 'acribus'

export default {
  schema: [
    new schema.Input('id', 'id', false, {
      indexed: true
    }),
    'name',
    'sex',
    'intro'
  ],
  url: '/person'
}

然后在 main.js 里引入

import resourcePerson from './resource/person'

// 注意： resource 声明 和 视图注册都需要放到 bootstrap 的回调里，确保执行时机正确
acribus.bootstrap(() => {
acribus.resource('person', resourcePerson)
})

如有其他资源，已同样的方式引入注册。

在项目下新建 view 文件夹， 用于存放所有的页面描述对象。 新建 profile.js:

export default {
  main: 'person',
  customHooks: [
    {
      label: 'View Log',
      handler () {
        //...
      }
    }
  ],
  actions: [
    {
      label: 'View Skill',
      handler (row) {
        // ...
      }
    }
  ]
}

然后在 main.js 里引入

import profile from './ view/profile.js'

acribus.bootstrap(() => {
acribus.resource('person', resourcePerson)

acribus.register(profile)
})

页面访问路径为主资源名 /person

参见 API 来查看所有接口说明。
