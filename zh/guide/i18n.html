# 多语言支持

# 使用自带的翻译功能

提示

目前 acribus 仅支持简体中文、繁体中文、英文。其他语言的支持已在计划中。 如果你需要使用其他的语言，欢迎贡献 PR：只需在 这里 添加一个语言配置文件即可。

要启用 arcibus 的多语言支持，首先需要使用如下的文件结构：

/
└─ /lang
   ├─ index.js
   ├─ zh-hant.json
   ├─ zh-hans.json
   └─ en.json

// 上文的 index.js
import en from './test-en.json'
import zhHans from './test-zh-Hans.json'
import zhHant from './test-zh-Hant.json'

// key 需要与 element 的 key 相同，以便于 element 的语言切换
export default {
  'en': { ...en, _name: 'English' },
  'zh-CN': { ...zhHans, _name: '简体中文' },
  'zh-TW': { ...zhHant, _name: '繁體中文' }
}

// 项目的 main.js
import acribus from 'acribus'
import locale from './lang/index'
// 如果 locale 是异步加载而来，应该使用 setI18nData，具体参见下面说明
Vue.use(acribus, {locale /* 上面定义的翻译信息 */})

在页面切换语言的地方引入辅助函数,切换时调用辅助函数更改语言设置即可：

import { setLocale } from 'arcibus'

// `locale` 为对应语言，比如 `en`/`zh-CN`/`zh-TW` 和 lang/index 里定义的键值一致
setLocale(locale)

在自定义页面引入 I18n 函数进行翻译

import { I18n } from 'acribus'

I18n('_name')

关于 I18n 的用法参见 辅助函数 I18n

如果需要异步加载翻译信息

import { setI18nData } from 'acribus'

// 在拉取数据成功的回调中书写
setI18nData(data)

# 使用其他 i18n 插件

如果不使用 acribus 自带的翻译功能，可以使用其他插件，比如 vue-i18n

import VueI18n from 'vue-i18n'
import langs from 'acribus/packages/lang'

Vue.use(acribus, {
  i18n: (key, value) => i18n.t(key, value)
})

Vue.use(VueI18n)

const messages = {
  en: {
    message: 'hello',
    ...langs.en // 或者用 Object.assign({ message: 'hello' }, langs.en)
  },
  zh: {
    message: '你好',
    ...langs['zh-CN'] // 或者用 Object.assign({ message: '你好' },langs['zh-CN'])
  }
}
// Create VueI18n instance with options
const i18n = new VueI18n({
  locale: 'en', // set locale
  messages, // set locale messages
})

Vue.use(acribus, {
  i18n: i18n.t
})

new Vue({ i18n }).$mount('#app')
